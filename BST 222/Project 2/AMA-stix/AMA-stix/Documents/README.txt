\title{BST 222, Project 2, Recurrence of paper "Association of Remnant Liver Ischemia With Early Recurrence
and Poor Survival After Liver Resection in Patients
With Hepatocellular Carcinoma"}

\author[1]{Bohao Zou}

\authormark{AUTHOR ONE \textsc{et al}}


\address[1]{\orgdiv{Statistic}, \orgname{UC Davis}, \orgaddress{\state{CA}, \country{USA}}}

\presentaddress{Davis, CA, USA}

\abstract[Summary]{In this project, we need to simulate the data, which came from the paper "\emph{Association of Remnant Liver Ischemia With Early Recurrence and Poor Survival After Liver Resection in Patients With Hepatocellular Carcinoma}" and we need to recur the result by using survival analysis. The main objective of original paper is "\emph{To evaluate whether remnant liver ischemia (RLI) may have an adverse effect on long-term survival and morbidity after liver resection in patients with hepatocellular carcinoma}". The main conclusion and relevance of the original paper is that "\emph{Preventive management and technical refinements in
hepatectomy are important to decrease the risk of RLI and to improve survival of patients
with hepatocellular carcinoma.}"}

\keywords{Survival analysis, Simulate data, SAS}
